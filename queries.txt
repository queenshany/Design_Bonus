********************** TRANSACTION REPORT QUERY **********************

select creationDate, status, SUM(C) as transAmount,  SUM(C) * 100/ SUM(P) as transPercent, Min(transMin) as trMin, MAX(TransMax) as trMax
FROM (SELECT creationDate, status, COUNT(*) as C, (Select COUNT(*) FROM tblTransConfirm WHERE creationDate = TC.creationDate) as P, MIN(commissionFee) as transMin, MAX(commissionFee) as TransMax
FROM tblTransConfirm TC 
group by  creationDate,status

UNION ALL

SELECT creationDate, status, COUNT(*) as C, (Select COUNT(*) FROM tblTransPay WHERE creationDate = TP.creationDate) as P, MIN(commissionFee) as transMin, MAX(commissionFee) as TransMax
From tblTransPay TP
group by creationDate, status)
group by creationDate, status
HAVING creationDate=[]
ORDER BY creationDate

********************** CALC PROBABILITY QUERY **********************

SELECT SUM(C) * 100/ SUM(P) AS transPercent
FROM (SELECT creationDate, status, COUNT(*) as C, (Select COUNT(*) FROM tblTransConfirm WHERE creationDate = TC.creationDate) as P
FROM tblTransConfirm TC 
group by  creationDate,status

UNION ALL

SELECT creationDate, status, COUNT(*) as C, (Select COUNT(*) FROM tblTransPay WHERE creationDate = TP.creationDate) as P
From tblTransPay TP
group by creationDate, status)  AS C
GROUP BY creationDate, status
HAVING creationDate=[] AND status="DONE"
ORDER BY creationDate;
